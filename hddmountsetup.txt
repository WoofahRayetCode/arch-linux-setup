#Get disk names
sudo blkid
sudo blkid | grep UUID=
#Make dirs to mount drives to
mkdir GAMES
#SSD UUIDs
LABEL="Games" UUID="cd6b63a7-7e97-4896-a116-6f58c35c6881"

#Open fstab file
sudo nano /etc/fstab
sudo gedit /etc/fstab

#Mount SSD
UUID=cd6b63a7-7e97-4896-a116-6f58c35c6881 /home/ericparsley/GAMES ext4 defaults 0 0
#Gain ownership of hdds and ssd
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/GAMES
