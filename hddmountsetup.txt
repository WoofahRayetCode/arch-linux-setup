#Get disk names
sudo blkid | grep UUID=

#Make dirs to mount drives to
mkdir [hddorssdname]
mkdir Storage

#Open fstab file
sudo nano /etc/fstab

[uuid=#] /home/[username/[hddname] [filesystem] defaults 0 0
#Mount STORAGE
UUID=78ae0b75-ed91-4d15-a8a1-55c360c64e17 /home/ericparsley/Storage btrfs defaults 0 0

#Reload to use next command
sudo systemctl daemon-reload

#Refresh stuff so I don't have to restart anymore :D
sudo mount -a

#Gain ownership of HDDs
sudo chown -R -v [username]:[username] /home/[username]/[hddname]
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Storage
