#Get disk names
sudo blkid
sudo blkid | grep UUID=
#Make dirs to mount drives to
mkdir GAMES
#SSD UUIDs
LABEL="Games" UUID="e8ab17d1-572e-41fb-ae20-b8ffaad5a1db"

#Open fstab file
sudo nano /etc/fstab
sudo gedit /etc/fstab

#Mount SSD
UUID=e8ab17d1-572e-41fb-ae20-b8ffaad5a1db /home/ericparsley/GAMES btrfs defaults 0 0
#Gain ownership of hdds and ssd
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/GAMES

#Mount External HDD
UUID=D6A6B87DA6B86023 /home/ericparsley/EXTERNAL auto defaults
#Get ownership of External HDD
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/EXTERNAL
