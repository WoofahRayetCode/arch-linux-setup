#Get disk names
sudo blkid
sudo blkid | grep UUID=
#Make dirs to mount drives to
mkdir GAMES
#SSD UUIDs
LABEL="Games" UUID="8f062953-f1c8-47b8-8828-7e7972639f5c"

#Open fstab file
sudo nano /etc/fstab
sudo gedit /etc/fstab

#Mount SSD
UUID=8f062953-f1c8-47b8-8828-7e7972639f5c /home/ericparsley/GAMES   ext4 defaults  0    0
#Gain ownership of hdds and ssd
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/GAMES
