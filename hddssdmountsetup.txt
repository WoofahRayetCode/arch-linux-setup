
#Get disk names
sudo blkid | grep UUID=

#Make dirs to mount drives to
mkdir Games Storage

#Open fstab file
sudo nano /etc/fstab

#Games SSD
UUID=18ba3f0a-47ca-467a-bfe9-fbc973b5884a /home/ericparsley/Games btrfs defaults 0 0

#Storage SSD
UUID=e41f0a87-207e-4b62-ad46-1455ea25b36b /home/ericparsley/Storage btrfs defaults 0 0

#Reload to use next command
sudo systemctl daemon-reload

#Refresh stuff so I don't have to restart anymore :D
sudo mount -a

#Gain ownership of HDDs/SSDs
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Games
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Storage
