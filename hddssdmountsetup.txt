#Get disk names
sudo blkid | grep UUID=

#Make dirs to mount drives to
mkdir Games Storage WDMyCloudNAS

#Mount NAS
sudo mount.cifs //192.168.0.183/Public /home/ericparsley/WDMyCloudNAS -o username=ericparsley,uid=1000,gid=1000,vers=2.0

#Open fstab file
sudo nano /etc/fstab

#Storage HDD
UUID=3ac25546-214a-403b-92d3-a834586e0a09 /home/ericparsley/Storage btrfs defaults 0 0

#Network Share Mount
//192.168.0.183/Public /home/ericparsley/WDMyCloudNAS cifs rw,auto,nofail,file_mode=0777,dir_mode=0777,umask=0 0 0

#Reload to use next command
sudo systemctl daemon-reload

#Refresh stuff so I don't have to restart anymore :D
sudo mount -a

#Gain ownership of HDDs/SSDs
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Storage
