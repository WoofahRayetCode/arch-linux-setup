
#Get disk names
sudo blkid | grep UUID=

#Make dirs to mount drives to
mkdir Data
mkdir ExtHDD

#Open fstab file
sudo nano /etc/fstab

#Data SSD Desktop
UUID=77eb65ee-a255-4d25-b1c4-1815b3f5a0d6 /home/gamers/Data btrfs defaults 0 0

#Data SSD Laptop
UUID=acdc5f50-ed9d-416e-8c29-ae91d91b761b /home/ericparsley/Data btrfs defaults 0 0

#External 5TB HDD
UUID=34B455AAB4556EFC /home/gamerscomp/ExtHDD ntfs defaults 0 0

#Reload to use next command
sudo systemctl daemon-reload

#Refresh stuff so I don't have to restart anymore :D
sudo mount -a

#Gain ownership of secondary drive on Laptop
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Data
#Gain ownership of secondary drive on Desktop
sudo chown -R -v gamers:gamers /home/gamers/Data
#Gain full ownership of External 5TB HDD for desktop
sudo chown -R -v gamers:gamers /home/gamers/ExtHDD
