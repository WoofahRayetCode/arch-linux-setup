#Get disk names
sudo blkid | grep UUID=

#Make dirs to mount drives to
mkdir Games Storage WDMyCloudNAS

#Mount NAS
sudo mount.cifs //192.168.0.183/Public /home/ericparsley/WDMyCloudNAS -o username=ericparsley,uid=1000,gid=1000,vers=2.0

#Open fstab file
sudo nano /etc/fstab

#Game SSD
UUID=d2efb19a-7727-48b2-8903-5e51d1ec56dc /home/ericparsley/Games btrfs defaults 0 0

#Storage HDD
UUID=d87f8ddb-8609-4bf3-8e16-cfb878285d14 /home/ericparsley/Storage btrfs defaults 0 0

#Network Share Mount
//192.168.0.183/Public /mnt/share/wdmycloudnas cifs guest,uid=1000 0 0

#Reload to use next command
sudo systemctl daemon-reload

#Refresh stuff so I don't have to restart anymore :D
sudo mount -a

#Gain ownership of HDDs/SSDs
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Games
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Storage
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/WDMyCloudNAS
