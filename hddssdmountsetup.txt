
#Get disk names
sudo blkid | grep UUID=

mkdir Games Storage

#Open fstab file
sudo nano /etc/fstab

#Games and Data
UUID=4408b00e-be00-46c8-a415-cd2809954e89 /home/ericparsley/Games btrfs defaults 0 0

#Reload to use next command
sudo systemctl daemon-reload

#Refresh stuff so I don't have to restart anymore :D
sudo mount -a

#Gain ownership of HDDs/SSDs
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Games
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Storage
