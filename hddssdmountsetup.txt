#Get disk names
sudo blkid | grep UUID=

#Make dirs to mount drives to
mkdir Games Storage

#Open fstab file
sudo nano /etc/fstab

#Game SSD
UUID=7e162d4f-482c-44ac-8338-c182195b35aa /home/ericparsley/Games ext4 defaults 0 0

#Storage HDD
UUID=233b34fb-8bb1-4b1c-b613-84df5ad73b20 /home/ericparsley/Storage ext4 defaults 0 0

#Reload to use next command
sudo systemctl daemon-reload

#Refresh stuff so I don't have to restart anymore :D
sudo mount -a

#Gain ownership of HDDs/SSDs
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Games
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Storage
