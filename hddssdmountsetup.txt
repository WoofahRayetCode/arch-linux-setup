#Get disk names
sudo blkid | grep UUID=

#Make dirs to mount drives to
mkdir Storage Games

#Open fstab file
sudo nano /etc/fstab

#Games SSD
UUID=2a4f4fc8-79f2-46c3-b7d7-35f75f8777a8 /home/ericparsley/Games btrfs defaults 0 0

#Storage HDD
UUID=3ac25546-214a-403b-92d3-a834586e0a09 /home/ericparsley/Storage btrfs defaults 0 0

#Reload to use next command
sudo systemctl daemon-reload

#Refresh stuff so I don't have to restart anymore :D
sudo mount -a

#Gain ownership of HDDs/SSDs
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Games
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Storage
