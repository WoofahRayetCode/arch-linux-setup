
#Get disk names
sudo blkid | grep UUID=

#Make dirs to mount drives to
mkdir Games Storage

#Open fstab file
sudo nano /etc/fstab

#Games SSD
UUID=fd3921c7-69fd-4d60-9f29-5250488f3c13 /home/ericparsley/Games btrfs defaults 0 0

#Storage SSD
UUID=783f05d2-0d07-4845-9e3f-56ceedf2fb4a /home/ericparsley/Storage btrfs defaults 0 0

#Reload to use next command
sudo systemctl daemon-reload

#Refresh stuff so I don't have to restart anymore :D
sudo mount -a

#Gain ownership of HDDs/SSDs
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Games
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/Storage
