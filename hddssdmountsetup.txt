#Get disk names
sudo blkid | grep UUID=

#Make dirs to mount drives to
mkdir GAMES

#Open fstab file
sudo nano /etc/fstab

#Mount Manjaro Game Storage
UUID=f08b4996-a709-4250-a502-6244fa109b46 /home/ericparsley/GAMES btrfs defaults 0 0

#Mount Windows OS Drive
UUID=BCD220B3D2207434 /home/ericparsley/WindowsOS ntfs-3g defaults 0 0

#Mount Windows Storage
UUID=0A50F78F50F77FAD /home/ericparsley/WindowsStorage ntfs-3g defaults 0 0

#Reload to use next command
sudo systemctl daemon-reload

#Refresh stuff so I don't have to restart anymore :D
sudo mount -a

#Gain ownership of HDDs/SSDs
sudo chown -R -v ericparsley:ericparsley /home/ericparsley/GAMES
