sudo blkid | grep UUID=

mkdir SHDD2

#Open fstab file
sudo nano /etc/fstab

UUID=63715c15-a0c5-46c5-ab22-6bd4d84ddf99 /home/ericparsley/NEONSTORAGE ext4 defaults 0 0

#Mount Shared HDD #1

#Mount Shared HDD #2

sudo systemctl daemon-reload
